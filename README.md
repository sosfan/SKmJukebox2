  *gb.gui.qt
  
  *gb.image
  
  *gb.settings
  
  *gb.panel_led
  
  *gb.easyencrypt
  
  *mplayer
  
  *Skmdbmgr
  
  *SkmVideoServer(orangepi opcional)
